/* ================= 5. 隐藏彩蛋逻辑 ================= */ let titleClicks = 0; let clickTimer = null; function triggerEasterEgg() { titleClicks++; // 如果停顿超过 1.5 秒没点，就重新计数 if (clickTimer) clearTimeout(clickTimer); clickTimer = setTimeout(() => { titleClicks = 0; }, 1500); // 狂点 5 次触发 if (titleClicks >= 5) { titleClicks = 0; // 重置计数器 // 1. 触发全屏地震特效 document.body.classList.add('shake'); setTimeout(() => document.body.classList.remove('shake'), 500); // 2. 弹出隐藏对话框 document.getElementById('easterEggModal').showModal(); } }
